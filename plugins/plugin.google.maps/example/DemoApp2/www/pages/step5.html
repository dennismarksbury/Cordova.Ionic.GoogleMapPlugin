Step5: Draw a circle

Drawing a circle is really easy. Just call map.addCircle();

map.addCircle({
  'center': result.latLng,
  'radius': 100, //100m
  'strokeColor' : '#AA00FF',
  'strokeWidth': 5,
  'fillColor' : '#880000'
}, onCircleAdded);
function onCircleAdded(circle) {
  setTimeout(function() {
    circle.setFillColor("lightgreen");
  }, 3000);
}
