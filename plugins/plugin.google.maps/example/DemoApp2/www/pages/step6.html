var GORYOKAKU = new plugin.google.maps.LatLng(41.797184,140.756831);
map.addPolyline({
  points: [
    result.latLng,
    GORYOKAKU
  ],
  'color' : '#AA00FF',
  'width': 10,
  'geodesic': true
}, onPolylineAdded);
function onPolylineAdded(polyline) {
  map.animateCamera({
    target: GORYOKAKU,
    zoom: 17,
    tilt: 50
